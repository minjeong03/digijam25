9static std::vector<char32_t> UTF8ToCodepoints(const std::string& text)
10{
11 std::vector<char32_t> result;
12 size_t i = 0;
13
14 while (i < text.size())
15 {
16 char32_t cp = 0;
17 unsigned char byte = text[i];
18
19 if ((byte & 0b10000000) == 0)
20 {
21 cp = byte;
22 ++i;
23 }
24 else if ((byte & 0b11100000) == 0b11000000)
25 {
26 cp = (byte & 0b00011111) << 6;
27 cp |= (text[i + 1] & 0b00111111);
28 i += 2;
29 }
30 else if ((byte & 0b11110000) == 0b11100000)
31 {
32 cp = (byte & 0b00001111) << 12;
33 cp |= (text[i + 1] & 0b00111111) << 6;
34 cp |= (text[i + 2] & 0b00111111);
35 i += 3;
36 }
37 else if ((byte & 0b11111000) == 0b11110000)
38 {
39 cp = (byte & 0b00000111) << 18;
40 cp |= (text[i + 1] & 0b00111111) << 12;
41 cp |= (text[i + 2] & 0b00111111) << 6;
42 cp |= (text[i + 3] & 0b00111111);
43 i += 4;
44 }
45 else
46 {
47 ++i;
48 continue;
49 }
50
51 result.push_back(cp);
52 }
53
54 return result;
55}

103bool Font::TryBakeGlyph(char32_t c)
104{

107

152 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);

156 GL_RED, GL_UNSIGNED_BYTE,
157 g->bitmap.buffer
158 );
159 glPixelStorei(GL_UNPACK_ALIGNMENT, prevAlignment);
160 }
161 else
162 {
163 unsigned char dummy = 0;

178 static_cast<float>(drawX) / atlasTexture->GetWidth(),
179 static_cast<float>(drawY) / atlasTexture->GetHeight()
180 };
181 glyph.uvBottomRight = {
182 static_cast<float>(drawX + safeW) / atlasTexture->GetWidth(),
183 static_cast<float>(drawY + safeH) / atlasTexture->GetHeight()

188 nextX += cellW;
189 maxRowHeight = std::max(maxRowHeight, cellH);

212{
213 float maxWidth = 0.0f;
214 float currentLineWidth = 0.0f;
215 int lineCount = 1;
216
217 std::vector<char32_t> codepoints = UTF8ToCodepoints(text);
218 for (char32_t c : codepoints)

220 if (c == U'\n')
