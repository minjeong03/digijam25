132void LoadingState::UploadToGPU(const LoadResult& r, const EngineContext& engineContext)
133{
134 std::visit([&](auto&& res)
135 {
136 using T = std::decay_t<decltype(res)>;
137 if constexpr (std::is_same_v<T, TextureResult>)
138 {
139 if (!res.ok)
140 return;
141
142 auto tex = std::make_unique<Texture>(res.pixels.data(), res.width, res.height, res.channels, res.settings);
143 engineContext.renderManager->RegisterTexture(res.tag, std::move(tex));
144 }
145 else if constexpr (std::is_same_v<T, ShaderResult>)
146 {
147 if (!res.ok)
148 return;
149
150 auto shader = std::make_unique<Shader>();
151 for (auto& s : res.sources)
152 {
153 shader->AttachFromSource(s.stage, s.source);
154 }
155 if (!shader->Link())
156 {
157 JIN_ERR("Shader link failed (tag=" << res.tag << ")");
158 return;
159 }
160 engineContext.renderManager->RegisterShader(res.tag, std::move(shader));
161 }
162 else if constexpr (std::is_same_v<T, FontResult>)
163 {
164 engineContext.renderManager->RegisterFont(res.tag, res.ttfPath, res.pixelSize);
165 }
166 else if constexpr (std::is_same_v<T, SoundResult>)
167 {
168 engineContext.soundManager->LoadSound(res.tag, res.filePath, res.loop);
169 }
170 }, r);
171}
