124void Texture::GenerateTexture(const unsigned char* data, const TextureSettings& settings)
125{
126 GLenum internalFormat = GL_RGBA8;
127 GLenum pixelFormat = GL_RGBA;
128
129 if (channels == 1)
130 {
131 internalFormat = GL_R8;
132 pixelFormat = GL_RED;
133 }
134 else if (channels == 3)
135 {
136 internalFormat = GL_RGB8;
137 pixelFormat = GL_RGB;
138 }
139 else if (channels == 4)
140 {
141 internalFormat = GL_RGBA8;
142 pixelFormat = GL_RGBA;
143 }
144
145 GLint prev = 0;
146 glGetIntegerv(GL_UNPACK_ALIGNMENT, &prev);
147 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
148 glCreateTextures(GL_TEXTURE_2D, 1, &id);
149 glTextureStorage2D(id, settings.generateMipmap ? 1 + floor(log2(std::max(width, height))) : 1, internalFormat, width, height);
150 if (data)
151 glTextureSubImage2D(id, 0, 0, 0, width, height, pixelFormat, GL_UNSIGNED_BYTE, data);
152 else
153 {
154 if (pixelFormat == GL_RGBA)
155 {
156 const GLuint zero[4] = { 0,0,0,0 };
157 glClearTexImage(id, 0, GL_RGBA, GL_UNSIGNED_INT, zero);
158 }
159 else if (pixelFormat == GL_RGB)
160 {
161 const GLuint zero[3] = { 0,0,0 };
162 glClearTexImage(id, 0, GL_RGB, GL_UNSIGNED_INT, zero);
163 }
164 }
165 glTextureParameteri(id, GL_TEXTURE_MIN_FILTER, ConvertMinFilter(settings.minFilter));
166 glTextureParameteri(id, GL_TEXTURE_MAG_FILTER, ConvertMagFilter(settings.magFilter));
167 glTextureParameteri(id, GL_TEXTURE_WRAP_S, ConvertWrap(settings.wrapS));
168 glTextureParameteri(id, GL_TEXTURE_WRAP_T, ConvertWrap(settings.wrapT));
169
170 if (settings.generateMipmap)
171 {
172 glGenerateTextureMipmap(id);
173 }
174 glPixelStorei(GL_UNPACK_ALIGNMENT, prev);
175}
