200void SoundManager::ControlByID(SoundControlType control, SoundInstanceID id)
201{
202 auto it = instanceMap.find(id);
203 if (it == instanceMap.end() || it->second == nullptr) return;
204
205 SoundInstanceMA* inst = it->second;
206
207 switch (control)
