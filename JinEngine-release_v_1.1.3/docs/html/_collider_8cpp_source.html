163{

185void SpatialHashGrid::Insert(Object* obj)
186{
187 if (!obj->IsAlive() || !obj->GetCollider())
188 return;
189 objects.push_back(obj);
190 Collider* collider = obj->GetCollider();
191 glm::vec2 offset = collider? collider->GetOffset(): glm::vec2(0);
192 glm::vec2 pos = obj->GetWorldPosition()+ offset;
193 float radius = obj->GetCollider()->GetBoundingRadius();
194
195 glm::ivec2 minCell = GetCell(pos - glm::vec2(radius));
196 glm::ivec2 maxCell = GetCell(pos + glm::vec2(radius));
197 int coveredCells = (maxCell.x - minCell.x + 1) * (maxCell.y - minCell.y + 1);
198 if (coveredCells > LARGE_OBJECT_THRESHOLD)
199 {
200 largeObjects.push_back(obj);
201 return;
202 }
203
204 for (int y = minCell.y; y <= maxCell.y; ++y)
205 {
206 for (int x = minCell.x; x <= maxCell.x; ++x)
207 {
208 InsertToCell(obj, { x, y });
209 }
210 }
